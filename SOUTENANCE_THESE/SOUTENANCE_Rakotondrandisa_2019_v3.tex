\begin{eqnarray*}\nonumber
\nabla\cdot \vec{u}&=&0, \\
 \frac{\partial \vec{u}}{\partial t} + {(\vec{u}\cdot\nabla) \vec{u}} +\nabla p -\frac{1}{Re} \nabla^2 \vec{u}
 - \colpaper{f_B(\theta) \, \vec{e}_y} &=& 0, \\
 \frac{\pl  \theta}{\pl t} + \nabla \cdot\left(  \theta \vec{u}\right) -
 \frac{K}{Pr \ \Rey} \nabla^2 \theta  &=& 0.
\label{eq-energ}
\end{eqnarray*}

\visible<2>{
{
\color{blue} \hbox{\rule[1ex]{\textwidth}{.01cm}} \vspace{-0.7cm}

  
  \begin{equation*}
\vec{x} \rightarrow \frac{\tilde{\vec{x}}}{h}, \,  \vec{u} \rightarrow \frac{\tilde{\vec{u}}}{V_{ref}}, \,  \theta = \frac{T-T_c}{\delta T}, \,  t \rightarrow \frac{V_{ref}}{h} t_{\varphi}, \,  \delta T = T_h - T_c,
\label{eq-adim}
 \end{equation*}
  
  \begin{equation*}
	\Rey = \frac{V_{ref} h}{\nu_{liq}} , \quad \Pr = \frac{\nu_{l}}{\alpha_{l}},\quad  \Ray = \frac{g \beta h^3 \delta T}{\nu_{l} 	\alpha_{l}}.

 Convection naturelle de l'eau :

^{(*)} \rho(T)=\rho_m \left(1 - w \left|T - T_m\right|^q\right),

	f_B(\theta) = \frac{\Ray}{\Prd \, \Rey^2} \frac{1}{\beta \delta T}\, \frac{\rho(\theta_f)-\rho(\theta)}{\rho(\theta_f)}

\end{equation*}
