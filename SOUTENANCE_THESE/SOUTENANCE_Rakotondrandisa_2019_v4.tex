\section{Résolution numérique des équations de Navier-Stokes-Boussinesq}

\begin{frame}
  \frametitle{Equations de Navier-Stokes-Boussinesq}\vspace{-0.1cm} 

\begin{block}{}
Conservation de la masse / conservation de la quantité de mouvement / conservation de l'énergie.
\end{block}
\vspace{-0.2cm} 
\begin{eqnarray*}\nonumber
\nabla\cdot \vec{u}&=&0, \\
 \frac{\partial \vec{u}}{\partial t} + {(\vec{u}\cdot\nabla) \vec{u}} +\nabla p -\frac{1}{\Rey} \nabla^2 \vec{u}
 - \colpaper{f_B(\theta) \, \vec{e}_y} &=& 0, \\
 \frac{\pl  \theta}{\pl t} + \nabla \cdot\left(  \theta \vec{u}\right) -
 \frac{K}{Pr \ \Rey} \nabla^2 \theta  &=& 0.
\label{eq-energ}
\end{eqnarray*}

 \frac{\partial \vec{u}}{\partial t} + {(\vec{u}\cdot\nabla) \vec{u}} +\nabla p -\frac{1}{Re} \nabla^2 \vec{u}
 - \colpaper{f_B(\theta) \, \vec{e}_y} &=& A\!(\theta) \vec u, \\
 \frac{\pl \left(C \theta\right)}{\pl t} + \nabla \cdot\left( C \theta \vec{u}\right) -
 \frac{K}{Pr \ \Rey} \nabla^2 \theta +  \frac{\pl \left(C S\right)}{\pl t} &=& 0.
\label{eq-energ}
\end{eqnarray*}
